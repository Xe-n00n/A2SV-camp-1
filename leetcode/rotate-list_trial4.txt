# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:

        
        if not head or k == 0:
            return head
        x = head
        length = 0
        while x:
            length += 1
            x = x.next

        k = k % length
        if k == 0:
            return head
        
        fast = head
        slow = head
        i = 0
        
        while i < k:
            fast = fast.next
            i += 1
        
        while fast and fast.next:
            slow = slow.next
            fast = fast.next
        
        fast.next = head
        head = slow.next
        slow.next = None

        return head
