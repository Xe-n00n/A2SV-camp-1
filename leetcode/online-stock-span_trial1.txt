class StockSpanner:

    def __init__(self):
        self.stack = []


    def next(self, price: int) -> int:
        pops = 1
        while self.stack and price >= self.stack[-1][0]:
            element = self.stack.pop()
            pops += element[1]

        self.stack.append((price, pops))

        return pops
