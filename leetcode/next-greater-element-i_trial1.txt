class Solution:
    def nextGreaterElement(self, nums1: List[int], nums2: List[int]) -> List[int]:
        stack = []
        nextGreater = {}
        for num in nums2:
            while stack and num > stack[-1]:
                small = stack.pop()

                nextGreater[small] = num
            
            stack.append(num)
        
        return [nextGreater.get(n, -1) for n in nums1]
