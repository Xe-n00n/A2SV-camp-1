import math
class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        low = 1
        high = max(nums)
        ans = high
        while low <= high:
            mid = (low + high) // 2

            current_sum = sum(math.ceil(n / mid) for n in nums)

            if current_sum <= threshold:
                ans = mid
                high = mid - 1
            else:
                low = mid + 1
            
        return ans
