from collections import deque

class Solution:
    def maxSlidingWindow(self, nums: List[int], k: int) -> List[int]:
        queue = deque() 
        res = []
        # 0 
        for i in range(k):
            while queue and nums[queue[-1]] < nums[i]:
                el = queue.pop()
            
            queue.append(i)
        res.append(nums[queue[0]])

        for i in range(k, len(nums)):
            
            while queue and nums[queue[-1]] < nums[i]:
                el = queue.pop()
            queue.append(i)
           
            while queue and queue[-1] - queue[0] >= k:
                queue.popleft()

            
            res.append(nums[queue[0]])

        return res
