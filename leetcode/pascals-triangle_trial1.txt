class Solution:
    def generate(self, numRows: int) -> List[List[int]]:
        if numRows == 1:
            return [[1]]
        
        curr_row = [1]
        result = self.generate(numRows-1)
        last_row = result[-1]
        for i in range(1,len(last_row)):
            curr_row.append(last_row[i-1]+ last_row[i])
        
        curr_row.append(1)
        result.append(curr_row)
        return result
