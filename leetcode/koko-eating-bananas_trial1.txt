class Solution:
    def minEatingSpeed(self, piles: List[int], h: int) -> int:
        piles.sort()

        left = 1
        right = max(piles)
        ans = right
        while left <= right:
            mid = (left + right)//2

            time = 0 
            for pile in piles:
                time += math.ceil(pile / mid)

            if time > h:
                left = mid + 1
            else:
                ans = mid
                right = mid - 1
        
        return ans
