# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def deleteDuplicates(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head
        
        prev = head
        x = head.next
        while x: 
            if prev.val == x.val:
                prev.next = x.next
                x = x.next
            else: 
                x = x.next
                prev = prev.next

        return head
