# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def removeNthFromEnd(self, head: Optional[ListNode], n: int) -> Optional[ListNode]:
        if not head:
            return None

        dummy = ListNode(-1,head)
        left = dummy
        right =  left
        i = 0
        while right and i < n+1:
            right = right.next
            i += 1 

        while right:
            right = right.next
            left = left.next
        
        left.next = left.next.next

        return dummy.next
