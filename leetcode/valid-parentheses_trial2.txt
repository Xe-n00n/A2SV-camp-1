class Solution:
    def isValid(self, s: str) -> bool:
        validation = {
            ']' : '[' , 
            '}': '{',
              ')': '('
        }
        stack = []
        for c in s: 
            if c in validation.values():
                stack.append(c)
            else:
                if not stack:
                    return False
                if validation[c] == stack[-1]:
                    stack.pop()
                else:
                    return False
        if not stack:
            return True
        else:
            return False
