# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:
        carry = 0 
        l3 = ListNode(0)
        dummy = ListNode(-1, l3)
        prev = dummy
        while l1 and l2:
            current = l1.val + l2.val + carry
            result = current % 10 
            carry = current // 10

            l3.val = result 
            l3.next = ListNode(0)
            prev = l3
            l3 = l3.next
            l1 = l1.next
            l2 = l2.next
        
        

        while l2:
            current = l2.val + carry 
            result = current % 10
            carry = current // 10
            l3.val = result
            l3.next = ListNode(0)
            prev = l3
            l3 = l3.next
            l2 = l2.next

        while l1:
            current = l1.val + carry 
            result = current % 10
            carry = current // 10
            l3.val = result
            l3.next = ListNode(0)
            prev = l3
            l3 = l3.next
            l1= l1.next


        if carry:
            l3.val = carry
        else: 
            prev.next = None
        
        return dummy.next
