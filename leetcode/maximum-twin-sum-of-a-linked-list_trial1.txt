# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def pairSum(self, head: Optional[ListNode]) -> int:
        if not head:
            return 0
        fast = head
        slow = head
        prev = None
        while fast and fast.next:
            fast = fast.next.next
            prev = slow
            slow = slow.next
        stack = []
        
        while slow:
            stack.append(slow)
            slow = slow.next

        # list 1 is head to the middle
        if prev:
            prev.next = None
        # list 2 is created now with this loop
        head2 = stack.pop()
        curr = head2
        while stack:
            newNode = stack.pop()
            curr.next = newNode
            curr = curr.next

        curr.next = None
        max_sum = 0
        while head and head2:
            max_sum = max(max_sum, head.val + head2.val)
            head = head.next
            head2 = head2.next
        if head:
            max_sum = max(max_sum, head.val)
        if head2:
            max_sum = max(max_sum, head2.val)
        return max_sum
