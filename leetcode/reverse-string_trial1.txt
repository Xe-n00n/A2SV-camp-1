class Solution:
    def reverseString(self, s: List[str]) -> None:
        """
        Do not return anything, modify s in-place instead.
        """

        def reverser(left, right):
            if left >= right:
                return 
            else:
                s[right], s[left] = s[left], s[right]
                right -= 1
                left += 1 
            reverser(left,right)    
        
        reverser(0,len(s)-1)
