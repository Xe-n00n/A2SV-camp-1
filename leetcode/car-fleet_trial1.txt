class Solution:
    def carFleet(self, target: int, position: list[int], speed: list[int]) -> int:
        cars = sorted(zip(position, speed), reverse=True)
        
        stack = []
        for p, s in cars:
            time = (target - p) / s
            # print(time)
            # print(stack)
            if not stack or time > stack[-1]:

                stack.append(time)
                
        return len(stack)
