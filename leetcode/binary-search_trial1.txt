class Solution:
    def search(self, nums: List[int], target: int) -> int:
        left = 0 
        right = len(nums) - 1
        while left <= right:
            current = (left + right) // 2
            if nums[current] == target:
                return current
            elif nums[current] > target:
                right = current - 1
            else:
                left = current + 1 
        
        return -1
