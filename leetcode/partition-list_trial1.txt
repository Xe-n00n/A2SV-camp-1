# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def partition(self, head: Optional[ListNode], x: int) -> Optional[ListNode]:
        if not head or not head.next:
            return head
        
        dummyBefore = ListNode(0)
        dummyAfter = ListNode(0)
        before = dummyBefore
        after = dummyAfter
        
        while head:
            if head.val < x:
                before.next = head
                before = before.next
            else:
                after.next = head
                after = after.next
            head = head.next

        before.next = dummyAfter.next
        after.next = None
        return dummyBefore.next
